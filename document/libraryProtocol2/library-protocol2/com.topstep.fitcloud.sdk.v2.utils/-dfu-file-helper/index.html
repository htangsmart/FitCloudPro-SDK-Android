How to use cancelSignal?

Coroutine:

job = lifecycleScope.launch(Dispatchers.IO) {
   val file = suspendCancellableCoroutine {
       val cancelSignal = object : Disposable {
           @Volatile
           private var isCanceled = false

           override fun dispose() {
               isCanceled = true
           }

           override fun isDisposed(): Boolean {
               return isCanceled
           }
       }
       it.invokeOnCancellation {
           cancelSignal.dispose()
       }
       try {
           it.resume(createEpoOtaFile(...,cancelSignal))//file result
       } catch (e: Exception) {
           it.resumeWithException(e)//error result
       }
   }

disposable = Single.create {
    val cancelSignal = object : Disposable {
        @Volatile
        private var isCanceled = false

        override fun dispose() {
            isCanceled = true
        }

        override fun isDisposed(): Boolean {
            return isCanceled
        }
    }
    it.setDisposable(cancelSignal)
    try {
        it.onSuccess(createEpoOtaFile(..., cancelSignal))
    } catch (e: Exception) {
        it.tryOnError(e)
    }
}.subscribeOn(Schedulers.io()).subscribe({
    //file result
}, {
    //error result
})
